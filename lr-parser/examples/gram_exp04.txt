Stmt       -> AssignStmt | ExprStmt

ExprStmt   -> Expr ;

Expr -> AdditiveExpr
AdditiveExpr -> MultiplicativeExpr AdditiveExprRest
AdditiveExprRest -> + MultiplicativeExpr AdditiveExprRest | - MultiplicativeExpr AdditiveExprRest | epsilon
MultiplicativeExpr -> Factor MultiplicativeExprRest
MultiplicativeExprRest -> * Factor MultiplicativeExprRest | / Factor MultiplicativeExprRest | epsilon
